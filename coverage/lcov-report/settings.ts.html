					this.plugin.settings.tagsExclude = value;
					await this.plugin.saveSettings();
				}));

						await this.plugin.saveSettings();
				}));
				
    containerEl.createEl('h2', { text: 'Other Settings' });

			.addToggle((toggle) => 
				toggle
